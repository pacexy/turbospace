# MDX

## Line numbers and line highlighting

```tsx showLineNumbers {4, 7-10}
import { useFloating } from '@floating-ui/react-dom'

function MyComponent() {
  const { x, y, reference, floating } = useFloating()

  return (
    <>
      <div ref={reference} />
      <div ref={floating} />
    </>
  )
}
```
